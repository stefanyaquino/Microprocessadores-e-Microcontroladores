Para todas as questões abaixo, utilize o modo de comparação do Timer A.

Para os itens abaixo, confira a diferença no brilho do LED. (a) Pisque o LED no pino P1.6 numa frequência de 100 Hz e ciclo de trabalho de 25%.
#include<msp430g2553.h>
#define LED BIT6
#define PERIODO 10000
#define DUTY_CICLE 2500

int main(void)
{
	WDTCTL = WDTPW + WDTHOLD;
	BCSCTL1 = CALBC1_1MHZ;
	DCOCTL = CALDCO_1MHZ;
	P1DIR |= LED;
	P1SEL |= LED;
	P1SEL2 &= ~LED;
	TACCR0 = PERIODO-1;
	TACCR1 = DUTY_CICLE-1; // ciclo em que o led permanece ativo
	TACCTL1 = OUTMOD_7;
	TACTL = TASSEL_2 + ID_0 + MC_1; // SMCLK Utilizado , dividido por 1, contagem completa
	_BIS_SR(LPM0_bits);
	return 0;
}
